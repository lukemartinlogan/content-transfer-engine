121  void SetupAdapterMode() {
122  char* adapter_mode_str = getenv(kAdapterMode);
123  if (adapter_mode_str == nullptr) {
124  // default is Persistent mode
125  adapter_mode = AdapterMode::kDefault;
126  } else {
127  if (strcmp(kAdapterDefaultMode, adapter_mode_str) == 0) {
128  adapter_mode = AdapterMode::kDefault;
129  } else if (strcmp(kAdapterBypassMode, adapter_mode_str) == 0) {
130  adapter_mode = AdapterMode::kBypass;
131  } else if (strcmp(kAdapterScratchMode, adapter_mode_str) == 0) {
132  adapter_mode = AdapterMode::kScratch;
133  } else if (strcmp(kAdapterWorkflowMode, adapter_mode_str) == 0) {
134  adapter_mode = AdapterMode::kWorkflow;
135  } else {
136  // TODO(hari): @errorhandling throw error.
137  return;
138  }
139  }
140  char* adapter_paths_str = getenv(kAdapterModeInfo);
141  std::vector<std::string> paths_local;
142  if (adapter_paths_str) {
143  paths_local = StringSplit(adapter_paths_str, kPathDelimiter);
144  }
145  adapter_paths = paths_local;
146  }

161  bool Persists(std::string path) {
162  if (adapter_mode == AdapterMode::kDefault ||
163  adapter_mode == AdapterMode::kWorkflow) {
164  if (adapter_paths.empty()) {
165  return true;
166  } else {
167  for (const auto& pth : adapter_paths) {
168  if (path.find(pth) == 0) {
169  return true;
170  }
171  }
172  return false;
173  }
174  } else if (adapter_mode == AdapterMode::kScratch) {
175  if (adapter_paths.empty()) {
176  return false;
177  } else {
178  for (const auto& pth : adapter_paths) {
179  if (path.find(pth) == 0) {
180  return false;
181  }
182  }
183  return true;
184  }
185  }
186  return true;
187  }
