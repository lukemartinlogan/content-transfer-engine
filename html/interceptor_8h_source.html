111  void SetupAdapterMode() {
112  char* adapter_mode_str = getenv(kAdapterMode);
113  if (adapter_mode_str == nullptr) {
114  // default is Persistent mode
115  adapter_mode = AdapterMode::kDefault;
116  } else {
117  if (strcmp(kAdapterDefaultMode, adapter_mode_str) == 0) {
118  adapter_mode = AdapterMode::kDefault;
119  } else if (strcmp(kAdapterBypassMode, adapter_mode_str) == 0) {
120  adapter_mode = AdapterMode::kBypass;
121  } else if (strcmp(kAdapterScratchMode, adapter_mode_str) == 0) {
122  adapter_mode = AdapterMode::kScratch;
123  } else {
124  // TODO(hari): @errorhandling throw error.
125  return;
126  }
127  }
128  char* adapter_paths_str = getenv(kAdapterModeInfo);
129  std::vector<std::string> paths_local;
130  if (adapter_paths_str) {
131  paths_local = StringSplit(adapter_paths_str, kPathDelimiter);
132  }
133  adapter_paths = paths_local;
134  }

136  bool Persists(FILE* fh) { return Persists(GetFilenameFromFP(fh)); }
137  bool Persists(int fd) { return Persists(GetFilenameFromFD(fd)); }

139  bool Persists(std::string path) {
140  if (adapter_mode == AdapterMode::kDefault) {
141  if (adapter_paths.empty()) {
142  return true;
143  } else {
144  for (const auto& pth : adapter_paths) {
145  if (path.find(pth) == 0) {
146  return true;
147  }
148  }
149  return false;
150  }
151  } else if (adapter_mode == AdapterMode::kScratch) {
152  if (adapter_paths.empty()) {
153  return false;
154  } else {
155  for (const auto& pth : adapter_paths) {
156  if (path.find(pth) == 0) {
157  return false;
158  }
159  }
160  return true;
161  }
162  }
163  return true;
164  }
