74  MPI_Info info;
75  int status = MPI_File_get_info(*fh, &info);
76  if (status != MPI_SUCCESS) {
77  LOG(ERROR) << "MPI_File_get_info on file handler failed." << std::endl;
78  }
79  const int kMaxSize = 0xFFF;
80  int flag;
81  char filename[kMaxSize] = {0};
