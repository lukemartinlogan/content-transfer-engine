67  void InitializeHermes(bool is_mpi = false) {
68  if (ref == 0) {
69  this->is_mpi = is_mpi;
70  char* hermes_config = getenv(kHermesConf);
71  char* hermes_client = getenv(kHermesClient);
72  char* async_flush_mode = getenv(kHermesAsyncFlush);
73 
74  if (async_flush_mode && async_flush_mode[0] == '1') {
75  global_flushing_mode = FlushingMode::kAsynchronous;
76  } else {
77  global_flushing_mode = FlushingMode::kSynchronous;
78  }
79 
80  if (this->is_mpi) {
81  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
82  MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
83 
84  if ((hermes_client && hermes_client[0] == '1') || comm_size > 1) {
85  hermes = hermes::InitHermesClient(hermes_config);
86  } else {
87  this->is_mpi = false;
88  hermes = hermes::InitHermesDaemon(hermes_config);
89  }
90  } else {
91  hermes = hermes::InitHermesDaemon(hermes_config);
92  }
93  INTERCEPTOR_LIST->SetupAdapterMode();
94  }
95  ref++;
96  }
101  void FinalizeHermes() {
102  if (ref == 1) {
103  if (this->is_mpi) {
104  MPI_Barrier(MPI_COMM_WORLD);
105  char *stop_daemon = getenv(kStopDaemon);
106  bool shutdown_daemon = true;
107  if (stop_daemon && stop_daemon[0] == '0') {
108  shutdown_daemon = false;
109  }
110  hermes->FinalizeClient(shutdown_daemon);
111  } else {
112  hermes->Finalize(true);
113  }
114  }
115  ref--;
116  }
